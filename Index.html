        // Navigazione e Rendering UI (Invariati)

        function goWallet() {

            updateUI();

        }

        function updateUI() {

            const list = document.getElementById('card-list');

            const fab = document.getElementById('add-fab');

            list.innerHTML = '';

           

            if(cards.length === 0) {

                fab.className = 'fab fab-center';

                list.innerHTML = '<p style="text-align:center; color:#555; margin-top:40px;">Nessuna carta presente.</p>';

            } else {

                fab.className = 'fab fab-bottom';

                cards.forEach((c, i) => {

                    const div = document.createElement('div');

                    div.className = 'card';

                    const imgSrc = c.front || `https://ui-avatars.com/api/?name=${encodeURIComponent(c.name)}&background=333&color=fff`;

                    div.innerHTML = `<img src="${imgSrc}" class="card-img"><div style="font-weight:600; font-size:18px;">${c.name}</div>`;

                    div.onclick = () => showDetail(i);

                    list.appendChild(div);

                });

            }

        }

        function openAddModal() {

            document.getElementById('in-name').value = '';

            document.getElementById('file-front').value = '';

            document.getElementById('file-back').value = '';

           

            const pF = document.getElementById('prev-f');

            const pB = document.getElementById('prev-b');

            pF.src = ''; pF.style.display = 'none';

            pB.src = ''; pB.style.display = 'none';

           

            document.getElementById('add-modal').style.display = 'block';

        }

        function closeAddModal() {

            stopScan();

        }

        function showDetail(index) {

            currentCardIndex = index;

            const c = cards[index];

            document.getElementById('d-name').innerText = c.name;

            document.getElementById('d-front').src = c.front || `https://ui-avatars.com/api/?name=${encodeURIComponent(c.name)}&background=333&color=fff`;

           

            const backImg = document.getElementById('d-back');

            if(c.back) { backImg.src = c.back; backImg.style.display = 'block'; }

            else { backImg.style.display = 'none'; }

            document.getElementById('detail-modal').style.display = 'block';

           

            try { JsBarcode("#d-barcode", c.number, { format: "CODE128", width: 2, height: 80, displayValue: true }); }

            catch(e) { console.log("Errore Barcode", e); }

        }

        function deleteCard() {

            if(confirm("Vuoi eliminare questa carta?")) {

                cards.splice(currentCardIndex, 1);

                localStorage.setItem('myCardsData', JSON.stringify(cards));

                document.getElementById('detail-modal').style.display = 'none';

                updateUI();

            }

        }

        // SALVATAGGIO CON TRY-CATCH PER PREVENIRE ERRORI DI QUOTA

        function saveCard() {

            const name = document.getElementById('in-name').value.trim();

            const number = document.getElementById('in-num').value.trim();

            const pF = document.getElementById('prev-f');

            const pB = document.getElementById('prev-b');

            if(!name || !number) { alert("Inserisci Nome e Codice!"); return; }

            // Aggiungiamo temporaneamente la carta all'array

            const newCard = {

                name: name,

                number: number,

                front: pF.style.display === 'block' ? pF.src : null,

                back: pB.style.display === 'block' ? pB.src : null

            };

            cards.push(newCard);

           

            try {

                // Proviamo a salvare in memoria

                localStorage.setItem('myCardsData', JSON.stringify(cards));

                closeAddModal();

                updateUI();

            } catch (e) {

                // Se la memoria è ancora piena, eliminiamo la carta dall'array e avvisiamo l'utente

                cards.pop();

                if (e.name === 'QuotaExceededError' || e.name === 'NS_ERROR_DOM_QUOTA_REACHED') {

                    alert("⚠️ Memoria Piena! Hai salvato troppe carte. Elimina qualche vecchia carta per fare spazio.");

                } else {

                    alert("Errore durante il salvataggio.");

                }

            }

        }

        // COMPRESSIONE IMMAGINI (Novità per risolvere il problema della memoria)

        function previewAndCompress(input, id) {

            if (input.files && input.files[0]) {

                const reader = new FileReader();

                reader.onload = function(event) {

                    const imgObj = new Image();

                    imgObj.onload = function() {

                        const canvas = document.createElement('canvas');

                        const MAX_SIZE = 800; // Riduce la risoluzione a un max di 800px

                        let width = imgObj.width;

                        let height = imgObj.height;

                        if (width > height) {

                            if (width > MAX_SIZE) {

                                height *= MAX_SIZE / width;

                                width = MAX_SIZE;

                            }

                        } else {

                            if (height > MAX_SIZE) {

                                width *= MAX_SIZE / height;

                                height = MAX_SIZE;

                            }

                        }

                       

                        canvas.width = width;

                        canvas.height = height;

                        const ctx = canvas.getContext('2d');

                        ctx.drawImage(imgObj, 0, 0, width, height);

                       

                        // Comprime in formato JPEG al 70% di qualità per risparmiare moltissimo spazio

                        const compressedDataUrl = canvas.toDataURL('image/jpeg', 0.7);

                       

                        const img = document.getElementById(id);

                        img.src = compressedDataUrl;

                        img.style.display = 'block';

                    };

                    imgObj.src = event.target.result;

                };

                reader.readAsDataURL(input.files[0]);

            }

        }

        // Scanner

        function startScan() {

            document.getElementById('reader-container').style.display = 'block';

            if (!html5QrCode) { html5QrCode = new Html5Qrcode("reader"); }

            html5QrCode.start(

                { facingMode: "environment" },

                { fps: 10, qrbox: { width: 250, height: 150 } },

                (text) => { document.getElementById('in-num').value = text; stopScan(); }

            ).catch(err => { alert("Consenti l'accesso alla fotocamera."); });

        }

        function stopScan() {

            if (html5QrCode && html5QrCode.isScanning) {

                html5QrCode.stop().then(() => { document.getElementById('reader-container').style.display = 'none'; });

            } else {

                document.getElementById('reader-container').style.display = 'none';

            }

        }
